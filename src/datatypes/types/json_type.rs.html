37pub const JSON_TYPE_NAME: &str = "Json";
38const JSON_PLAIN_FIELD_NAME: &str = "__plain__";
39const JSON_PLAIN_FIELD_METADATA_KEY: &str = "is_plain_json";
40
41#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize, Default)]
42pub enum JsonFormat {
43    #[default]
44    Jsonb,
45    Native(Box<ConcreteDataType>),
46}
47
48/// JsonType is a data type for JSON data. It is stored as binary data of jsonb format.
49/// It utilizes current binary value and vector implementation.
50#[derive(Debug, Default, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
51pub struct JsonType {
52    pub format: JsonFormat,
53}
54
55impl JsonType {
56    pub fn new(format: JsonFormat) -> Self {
57        Self { format }
58    }
59
60    pub(crate) fn new_native(native: ConcreteDataType) -> Self {
61        Self {
62            format: JsonFormat::Native(Box::new(native)),
63        }
64    }
65
66    pub(crate) fn native_type(&self) -> ConcreteDataType {
67        match &self.format {
68            JsonFormat::Jsonb => ConcreteDataType::binary_datatype(),
69            JsonFormat::Native(x) => x.as_ref().clone(),
70        }
71    }
72
73    pub(crate) fn empty() -> Self {
74        Self {
75            format: JsonFormat::Native(Box::new(ConcreteDataType::null_datatype())),
76        }
77    }
78
79    /// Make json type a struct type, by:
80    /// - if the json is an object, its entries are mapped to struct fields, obviously;
81    /// - if not, the json is one of bool, number, string or array, make it a special field
82    ///   (see [plain_json_struct_type]).
83    pub(crate) fn as_struct_type(&self) -> StructType {
84        match &self.format {
85            JsonFormat::Jsonb => StructType::default(),
86            JsonFormat::Native(inner) => match inner.as_ref() {
87                ConcreteDataType::Struct(t) => t.clone(),
88                x => plain_json_struct_type(x.clone()),
89            },
90        }
91    }
92
93    /// Check if this json type is the special "plain" one.
94    /// See [JsonType::as_struct_type].
95    #[expect(unused)]
96    pub(crate) fn is_plain_json(&self) -> bool {
97        let JsonFormat::Native(box ConcreteDataType::Struct(t)) = &self.format else {
98            return true;
99        };
100        let fields = t.fields();
101        let Some((single, [])) = fields.split_first() else {
102            return false;
103        };
104        single.name() == JSON_PLAIN_FIELD_NAME
105            && single.metadata(JSON_PLAIN_FIELD_METADATA_KEY) == Some("true")
106    }
107
108    /// Try to merge this json type with others, error on datatype conflict.
109    pub(crate) fn merge(&mut self, other: &JsonType) -> Result<()> {
110        match (&self.format, &other.format) {
111            (JsonFormat::Jsonb, JsonFormat::Jsonb) => Ok(()),
112            (JsonFormat::Native(this), JsonFormat::Native(that)) => {
113                let merged = merge(this.as_ref(), that.as_ref())?;
114                self.format = JsonFormat::Native(Box::new(merged));
115                Ok(())
116            }
117            _ => MergeJsonDatatypeSnafu {
118                reason: "json format not match",
119            }
120            .fail(),
121        }
122    }
123
124    pub(crate) fn is_mergeable(&self, other: &JsonType) -> bool {
125        match (&self.format, &other.format) {
126            (JsonFormat::Jsonb, JsonFormat::Jsonb) => true,
127            (JsonFormat::Native(this), JsonFormat::Native(that)) => {
128                is_mergeable(this.as_ref(), that.as_ref())
129            }
130            _ => false,
131        }
132    }
133}

135/// A special struct type for denoting "plain"(not object) json value. It has only one field, with
136/// fixed name [JSON_PLAIN_FIELD_NAME] and with metadata [JSON_PLAIN_FIELD_METADATA_KEY] = `"true"`.
137pub(crate) fn plain_json_struct_type(item_type: ConcreteDataType) -> StructType {
138    let mut field = StructField::new(JSON_PLAIN_FIELD_NAME.to_string(), item_type, true);
139    field.insert_metadata(JSON_PLAIN_FIELD_METADATA_KEY, true);
140    StructType::new(Arc::new(vec![field]))
141}

143fn is_mergeable(this: &ConcreteDataType, that: &ConcreteDataType) -> bool {
144    fn is_mergeable_struct(this: &StructType, that: &StructType) -> bool {
145        let this_fields = this.fields();
146        let this_fields = this_fields
147            .iter()
148            .map(|x| (x.name(), x))
149            .collect::<HashMap<_, _>>();
150
151        for that_field in that.fields().iter() {
152            if let Some(this_field) = this_fields.get(that_field.name())
153                && !is_mergeable(this_field.data_type(), that_field.data_type())
154            {
155                return false;
156            }
157        }
158        true
159    }
160
161    match (this, that) {
162        (this, that) if this == that => true,
163        (ConcreteDataType::List(this), ConcreteDataType::List(that)) => {
164            is_mergeable(this.item_type(), that.item_type())
165        }
166        (ConcreteDataType::Struct(this), ConcreteDataType::Struct(that)) => {
167            is_mergeable_struct(this, that)
168        }
169        (ConcreteDataType::Null(_), _) | (_, ConcreteDataType::Null(_)) => true,
170        _ => false,
171    }
172}
173
174fn merge(this: &ConcreteDataType, that: &ConcreteDataType) -> Result<ConcreteDataType> {
175    match (this, that) {
176        (this, that) if this == that => Ok(this.clone()),
177        (ConcreteDataType::List(this), ConcreteDataType::List(that)) => {
178            merge_list(this, that).map(ConcreteDataType::List)
179        }
180        (ConcreteDataType::Struct(this), ConcreteDataType::Struct(that)) => {
181            merge_struct(this, that).map(ConcreteDataType::Struct)
182        }
183        (ConcreteDataType::Null(_), x) | (x, ConcreteDataType::Null(_)) => Ok(x.clone()),
184        _ => MergeJsonDatatypeSnafu {
185            reason: format!("datatypes have conflict, this: {this}, that: {that}"),
186        }
187        .fail(),
188    }
189}
190
191fn merge_list(this: &ListType, that: &ListType) -> Result<ListType> {
192    let merged = merge(this.item_type(), that.item_type())?;
193    Ok(ListType::new(Arc::new(merged)))
194}
195
196fn merge_struct(this: &StructType, that: &StructType) -> Result<StructType> {
197    let this = Arc::unwrap_or_clone(this.fields());
198    let that = Arc::unwrap_or_clone(that.fields());
199
200    let mut this: BTreeMap<String, StructField> = this
201        .into_iter()
202        .map(|x| (x.name().to_string(), x))
203        .collect();
204    // merge "that" into "this" directly:
205    for that_field in that {
206        let field_name = that_field.name().to_string();
207        if let Some(this_field) = this.get(&field_name) {
208            let merged_field = StructField::new(
209                field_name.clone(),
210                merge(this_field.data_type(), that_field.data_type())?,
211                true, // the value in json object must be always nullable
212            );
213            this.insert(field_name, merged_field);
214        } else {
215            this.insert(field_name, that_field);
216        }
217    }
218
219    let fields = this.into_values().collect::<Vec<_>>();
220    Ok(StructType::new(Arc::new(fields)))
221}

223impl DataType for JsonType {
224    fn name(&self) -> String {
225        match &self.format {
226            JsonFormat::Jsonb => JSON_TYPE_NAME.to_string(),
227            JsonFormat::Native(x) => format!("Json<{x}>"),
228        }
229    }
230
231    fn logical_type_id(&self) -> LogicalTypeId {
232        LogicalTypeId::Json
233    }
234
235    fn default_value(&self) -> Value {
236        Bytes::default().into()
237    }
238
239    fn as_arrow_type(&self) -> ArrowDataType {
240        match self.format {
241            JsonFormat::Jsonb => ArrowDataType::Binary,
242            JsonFormat::Native(_) => ArrowDataType::Struct(Fields::empty()),
243        }
244    }
245
246    fn create_mutable_vector(&self, capacity: usize) -> Box<dyn MutableVector> {
247        match self.format {
248            JsonFormat::Jsonb => Box::new(BinaryVectorBuilder::with_capacity(capacity)),
249            JsonFormat::Native(_) => Box::new(JsonVectorBuilder::with_capacity(capacity)),
250        }
251    }
252
253    fn try_cast(&self, from: Value) -> Option<Value> {
254        match from {
255            Value::Binary(v) => Some(Value::Binary(v)),
256            _ => None,
257        }
258    }
259}

261/// Converts a json type value to string
262pub fn jsonb_to_string(val: &[u8]) -> Result<String> {
263    match jsonb::from_slice(val) {
264        Ok(jsonb_value) => {
265            let serialized = jsonb_value.to_string();
266            Ok(serialized)
267        }
268        Err(e) => InvalidJsonbSnafu { error: e }.fail(),
269    }
270}
271
272/// Converts a json type value to serde_json::Value
273pub fn jsonb_to_serde_json(val: &[u8]) -> Result<serde_json::Value> {
274    let json_string = jsonb_to_string(val)?;
275    serde_json::Value::from_str(json_string.as_str())
276        .context(DeserializeSnafu { json: json_string })
277}
278
279/// Parses a string to a json type value
280pub fn parse_string_to_jsonb(s: &str) -> Result<Vec<u8>> {
281    jsonb::parse_value(s.as_bytes())
282        .map_err(|_| InvalidJsonSnafu { value: s }.build())
283        .map(|json| json.to_vec())
284}

286#[cfg(test)]
287mod tests {
288    use super::*;
289    use crate::json::JsonStructureSettings;
290
291    #[test]
292    fn test_merge_json_type() -> Result<()> {
293        fn test(
294            json: &str,
295            json_type: &mut JsonType,
296            expected: std::result::Result<&str, &str>,
297        ) -> Result<()> {
298            let json: serde_json::Value = serde_json::from_str(json).unwrap();
299
300            let settings = JsonStructureSettings::Structured(None);
301            let value = settings.encode(json)?;
302            let value_type = value.data_type();
303            let Some(other) = value_type.as_json() else {
304                unreachable!()
305            };
306
307            let result = json_type.merge(other);
308            match (result, expected) {
309                (Ok(()), Ok(expected)) => {
310                    assert_eq!(json_type.name(), expected);
311                    assert!(json_type.is_mergeable(other));
312                }
313                (Err(err), Err(expected)) => {
314                    assert_eq!(err.to_string(), expected);
315                    assert!(!json_type.is_mergeable(other));
316                }
317                _ => unreachable!(),
318            }
319            Ok(())
320        }
321
322        let json_type = &mut JsonType::new(JsonFormat::Native(Box::new(
323            ConcreteDataType::null_datatype(),
324        )));
325
326        // can merge with json object:
327        let json = r#"{
328            "hello": "world",
329            "list": [1, 2, 3],
330            "object": {"a": 1}
331        }"#;
332        let expected =
333            r#"Json<Struct<"hello": String, "list": List<Int64>, "object": Struct<"a": Int64>>>"#;
334        test(json, json_type, Ok(expected))?;
335
336        // cannot merge with other non-object json values:
337        let jsons = [r#""s""#, "1", "[1]"];
338        let expects = [
339            r#"Failed to merge JSON datatype: datatypes have conflict, this: Struct<"hello": String, "list": List<Int64>, "object": Struct<"a": Int64>>, that: String"#,
340            r#"Failed to merge JSON datatype: datatypes have conflict, this: Struct<"hello": String, "list": List<Int64>, "object": Struct<"a": Int64>>, that: Int64"#,
341            r#"Failed to merge JSON datatype: datatypes have conflict, this: Struct<"hello": String, "list": List<Int64>, "object": Struct<"a": Int64>>, that: List<Int64>"#,
342        ];
343        for (json, expect) in jsons.into_iter().zip(expects.into_iter()) {
344            test(json, json_type, Err(expect))?;
345        }
346
347        // cannot merge with other json object with conflict field datatype:
348        let json = r#"{
349            "hello": 1,
350            "float": 0.123,
351            "no": 42
352        }"#;
353        let expected =
354            r#"Failed to merge JSON datatype: datatypes have conflict, this: String, that: Int64"#;
355        test(json, json_type, Err(expected))?;
356
357        // can merge with another json object:
358        let json = r#"{
359            "hello": "greptime",
360            "float": 0.123,
361            "int": 42
362        }"#;
363        let expected = r#"Json<Struct<"float": Float64, "hello": String, "int": Int64, "list": List<Int64>, "object": Struct<"a": Int64>>>"#;
364        test(json, json_type, Ok(expected))?;
365
366        // can merge with some complex nested json object:
367        let json = r#"{
368            "list": [4],
369            "object": {"foo": "bar", "l": ["x"], "o": {"key": "value"}},
370            "float": 0.456,
371            "int": 0
372        }"#;
373        let expected = r#"Json<Struct<"float": Float64, "hello": String, "int": Int64, "list": List<Int64>, "object": Struct<"a": Int64, "foo": String, "l": List<String>, "o": Struct<"key": String>>>>"#;
374        test(json, json_type, Ok(expected))?;
375
376        Ok(())
377    }
